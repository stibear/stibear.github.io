CLで解いているのですが、思わぬところで躓いてしまってます…

	(defun primep (n)
	  (cond
		((= 2 n) t)
		((= 3 n) t)
		((evenp n) nil)
		(t (loop for i from 3 to (isqrt n) by 2
			     never (zerop (mod n i))))))
	
	(defun furui (list)
	  (delete-if-not #'primep list))
